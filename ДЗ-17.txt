using System;

class Program
{
    static void Main()
    {
        Random random = new Random();
        int number = random.Next(1, 101); 
        Console.WriteLine($"Заданное число: {number}");

        int power = 0;        
        int result = 1;       

       
        while (result <= number)
        {
            result *= 2; 
            power++;      
        }

        Console.WriteLine($"Минимальная степень двойки: 2^{power} = {result}");
        Console.WriteLine($"Проверка: {number} < {result}");
